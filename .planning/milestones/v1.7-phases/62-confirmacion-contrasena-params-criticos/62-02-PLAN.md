            {/* Header */}
            <h2 className="text-lg font-bold text-[#4A5D3A] mb-2">
              Esta acción modifica un parámetro legal crítico

            {/* Body */}
            <p className="text-sm text-zinc-700 mb-1">

            <p className="text-sm text-zinc-600 mb-4">

**Hook API contract:**

```typescript
interface UsePasswordConfirmationOptions {
  onSuccess?: () => void;  // called after the wrapped action resolves successfully
}

interface UsePasswordConfirmationReturn {
  // Modal state — spread these as props onto <PasswordConfirmModal />
  isOpen: boolean;
  isLoading: boolean;
  error: string | null;
  paramName: string;
  // Modal control
  onConfirm: (password: string) => void;
  onCancel: () => void;
  // Gate function — call this instead of calling the action directly
  requireConfirmation: (paramName: string, action: (password: string) => Promise<void>) => void;
}
```

**Full implementation:**

```typescript
'use client';

import { useState, useCallback } from 'react';

interface UsePasswordConfirmationOptions {
  onSuccess?: () => void;
}

export function usePasswordConfirmation(options: UsePasswordConfirmationOptions = {}) {
  const [isOpen, setIsOpen] = useState(false);
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const [paramName, setParamName] = useState('');
  const [pendingAction, setPendingAction] = useState<((password: string) => Promise<void>) | null>(null);

  const requireConfirmation = useCallback(
    (name: string, action: (password: string) => Promise<void>) => {
      setParamName(name);
      setError(null);
      setPendingAction(() => action);
      setIsOpen(true);
    },
    []
  );

  const onConfirm = useCallback(
    async (password: string) => {
      if (!pendingAction) return;
      setIsLoading(true);
      setError(null);
      try {
        await pendingAction(password);
        setIsOpen(false);
        setPendingAction(null);
        options.onSuccess?.();
      } catch (err: unknown) {
        // Surface the backend error message if available (403 returns { error: string })
        const message =
          err instanceof Error
            ? err.message
            : 'Contraseña incorrecta. El cambio no fue guardado.';
        setError(message);
        // Do NOT close the modal — user can retry
      } finally {
        setIsLoading(false);
      }
    },
    [pendingAction, options]
  );

  const onCancel = useCallback(() => {
    if (isLoading) return; // prevent cancel while request is in flight
    setIsOpen(false);
    setError(null);
    setPendingAction(null);
  }, [isLoading]);

  return {
    isOpen,
    isLoading,
    error,
    paramName,
    onConfirm,
    onCancel,
    requireConfirmation,
  };
}
```

IMPORTANT implementation notes:
- `pendingAction` stores the async action function passed by the caller. This is the `(password: string) => Promise<void>` that the caller constructs — typically calling `legalParamService.patchParam({ ...dto, confirmationPassword: password })`.
- When the action throws, the error is captured and set on the modal WITHOUT closing it. The modal stays open and the user can retry (or cancel).
- `onCancel` is a no-op when `isLoading=true` to prevent closing while the bcrypt request is in flight.
- The hook does NOT import PasswordConfirmModal — it only manages state. The consuming page/component mounts PasswordConfirmModal and spreads the hook's return values as props.
- All async operations are wrapped in `useCallback` per project convention.
