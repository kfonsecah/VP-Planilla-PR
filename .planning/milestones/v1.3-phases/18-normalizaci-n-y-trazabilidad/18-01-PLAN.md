    const inTypes = ['in', 'entrada', 'entry', 'start', 'check_in', 'checkin'];
    const outTypes = ['out', 'salida', 'exit', 'end', 'check_out', 'checkout', 'salida final', 'fin turno'];

    if (outTypes.includes(v)) return 'OUT';
