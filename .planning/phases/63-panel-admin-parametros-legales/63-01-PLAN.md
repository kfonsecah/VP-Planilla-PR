```typescript
  /**
   * Get all active parameters at a given date, across all categories.
   * Deduplicates by key — only the most recent validFrom record per key is included.
   * @param date - Target date; defaults to today
   * @returns Array of active VpgLegalParam records
   */
  static async getActiveParams(date: Date = new Date()): Promise<VpgLegalParam[]> {
    const allParams = await prisma.vpgLegalParam.findMany({
      where: {
        validFrom: { lte: date },
        isActive: true,
      },
      orderBy: { validFrom: 'desc' },
    });

    const recent: Record<string, VpgLegalParam> = {};
    for (const param of allParams) {
      if (!recent[param.key]) {
        recent[param.key] = param as VpgLegalParam;
      }
    }
    return Object.values(recent);
  }
```

En `LegalParamController.ts`, añadir:
```typescript
  /**
   * GET /api/legal-params/active?date=YYYY-MM-DD
   * Returns all active parameters at the given date, across all categories.
   */
  static async getActiveParams(req: Request, res: Response): Promise<void> {
    const { date } = req.query;
    const targetDate = date ? new Date(date as string) : new Date();
    const params = await LegalParamService.getActiveParams(targetDate);
    res.status(200).json({ success: true, data: params });
  }
```

```typescript
  /**
   * Bulk updates minimum wages. Iterates over updates and calls upsertParam internally
   * within a single transaction. Since upsertParam now uses its own tx, we must rewrite
   * bulk update to use a shared transaction or do it manually. Actually, to keep it simple
   * and reuse logic, we will do sequential Prisma calls or extract the inner logic. 
   * Wait, upsertParam handles its own `$transaction`. We will just call `upsertParam` in a `Promise.all` or sequential loop since nesting transactions isn't natively supported the same way without passing the `tx` object. For simplicity, iterate sequentially.
   */
  static async bulkUpsertMinWages(
    updates: { key: string; value: number }[],
    validFrom: Date,
    source_decree: string,
    userId: string,
    passwordVerified: boolean = false
  ): Promise<VpgLegalParam[]> {
    const results: VpgLegalParam[] = [];
    for (const update of updates) {
      // Find current to get description and isCritical
      const current = await this.getParamAtDate(update.key, new Date());
      const description = current?.description || `Salario Mínimo ${update.key}`;
      const isCritical = current?.isCritical ?? false;
      
      const newParam = await this.upsertParam({
        key: update.key,
        value: new Decimal(update.value),
        description,
        category: 'MIN_WAGE',
        validFrom,
        isCritical,
        source_decree
      }, userId, { passwordVerified });
      results.push(newParam);
    }
    return results;
  }
```

En `LegalParamController.ts`, añadir:

```typescript
  /**
   * POST /api/legal-params/min-wages/bulk
   * Admin-only. Bulk updates minimum wages.
   */
  static async bulkUpsertMinWages(req: Request, res: Response): Promise<void> {
    const userId = String((req as any).user?.id ?? '');
    if (!userId) {
      res.status(401).json({ success: false, error: 'Unauthorized' });
      return;
    }

    const { updates, validFrom, source_decree, confirmationPassword } = req.body;
    if (!updates || !Array.isArray(updates) || !validFrom || !source_decree) {
      res.status(400).json({ success: false, error: 'Missing required fields: updates, validFrom, source_decree' });
      return;
    }

    // Require password confirmation for bulk update
    if (!confirmationPassword) {
      res.status(400).json({ success: false, error: 'La actualización masiva requiere confirmación de contraseña' });
      return;
    }
    const verified = await AuthService.verifyPasswordForUser(userId, confirmationPassword);
    if (!verified) {
      res.status(403).json({ success: false, error: 'Contraseña incorrecta. El cambio masivo no fue guardado.' });
      return;
    }

    const newParams = await LegalParamService.bulkUpsertMinWages(
      updates,
      new Date(validFrom),
      source_decree,
      userId,
      true // passwordVerified
    );
    res.status(201).json({ success: true, data: newParams });
  }
```
