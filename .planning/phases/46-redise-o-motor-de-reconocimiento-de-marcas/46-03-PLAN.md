import { prisma } from '../lib/prisma';

interface SuggestionResult {
  suggestedType: 'IN' | 'OUT';
  suggestedTime: string;
  confidence: 'HIGH' | 'MEDIUM' | 'LOW';
  reason: string;
}

export class MarkSuggestionService {
  /**
   * Analyzes employee historical marks to suggest a missing IN or OUT mark.
   * Returns null if no clear pattern found.
   */
  static async suggestMissingMark(
    employeeId: number,
    date: string,
    existingMarks: Array<{ type: string; time: string }>
  ): Promise<SuggestionResult | null> {
    // Get last 30 days of marks for this employee
    const startDate = new Date(date);
    startDate.setDate(startDate.getDate() - 30);
    
    const historicalMarks = await prisma.vpgClockLog.findMany({
      where: {
        employee_id: employeeId,
        clock_timestamp: { gte: startDate },
        clock_log_active: true,
      },
      orderBy: { clock_timestamp: 'desc' },
      take: 100,
    });

    if (historicalMarks.length < 5) {
      return null; // Not enough data for pattern
    }

    // Determine what mark type is missing based on existing marks for this day
    const dayMarks = existingMarks.filter(m => m.date === date);
    const hasMorningIn = dayMarks.some(m => m.type === 'IN' && m.hour < 12);
    const hasMorningOut = dayMarks.some(m => m.type === 'OUT' && m.hour < 12);
    const hasAfternoonIn = dayMarks.some(m => m.type === 'IN' && m.hour >= 12);
    const hasAfternoonOut = dayMarks.some(m => m.type === 'OUT' && m.hour >= 12);

    // Calculate average times for each mark type from history
    const morningEntries = historicalMarks.filter(m => m.clock_type === 'IN' && new Date(m.clock_timestamp).getHours() < 12);
    const morningExits = historicalMarks.filter(m => m.clock_type === 'OUT' && new Date(m.clock_timestamp).getHours() < 12);
    const afternoonEntries = historicalMarks.filter(m => m.clock_type === 'IN' && new Date(m.clock_timestamp).getHours() >= 12);
    const afternoonExits = historicalMarks.filter(m => m.clock_type === 'OUT' && new Date(m.clock_timestamp).getHours() >= 12);

    // Determine what needs to be suggested
    let suggestedType: 'IN' | 'OUT' | null = null;
    let timePool: typeof morningEntries = [];

    if (!hasMorningIn) {
      suggestedType = 'IN';
      timePool = morningEntries;
    } else if (!hasMorningOut) {
      suggestedType = 'OUT';
      timePool = morningExits;
    } else if (!hasAfternoonIn) {
      suggestedType = 'IN';
      timePool = afternoonEntries;
    } else if (!hasAfternoonOut) {
      suggestedType = 'OUT';
      timePool = afternoonExits;
    }

    if (!suggestedType || timePool.length === 0) {
      return null;
    }

    // Calculate average time from pool
    const avgMinutes = timePool.reduce((acc, m) => {
      const d = new Date(m.clock_timestamp);
      return acc + d.getHours() * 60 + d.getMinutes();
    }, 0) / timePool.length;

    const avgHours = Math.floor(avgMinutes / 60);
    const avgMins = Math.floor(avgMinutes % 60);
    const suggestedTime = `${avgHours.toString().padStart(2, '0')}:${avgMins.toString().padStart(2, '0')}`;

    // Determine confidence based on data consistency
    const variance = timePool.reduce((acc, m) => {
      const d = new Date(m.clock_timestamp);
      const mins = d.getHours() * 60 + d.getMinutes();
      return acc + Math.abs(mins - avgMinutes);
    }, 0) / timePool.length;

    const confidence = variance < 15 ? 'HIGH' : variance < 30 ? 'MEDIUM' : 'LOW';

    return {
      suggestedType,
      suggestedTime,
      confidence,
      reason: `Basado en ${timePool.length} marcas históricas similares`,
    };
  }
}

import { Request, Response } from 'express';
import { MarkSuggestionService } from '../service/MarkSuggestionService';

export class MarkSuggestionController {
  static async suggest(req: Request, res: Response) {
    const { employeeId, date } = req.body;
    const existingMarks = req.body.existingMarks || [];
    
    const suggestion = await MarkSuggestionService.suggestMissingMark(
      employeeId,
      date,
      existingMarks
    );
    
    res.json({ suggestion });
  }
}
