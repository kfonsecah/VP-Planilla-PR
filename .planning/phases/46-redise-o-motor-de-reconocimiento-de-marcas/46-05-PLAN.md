import React, { useState } from 'react';
import { useClockLogAdjustment } from '../hooks/useClockLogAdjustment';

interface MarkTypeSelectorProps {
  clockLogId: number;
  currentType: 'IN' | 'OUT';
  onChange?: () => void;
}

export function MarkTypeSelector({ clockLogId, currentType, onChange }: MarkTypeSelectorProps) {
  const [isOpen, setIsOpen] = useState(false);
  const { updateAdjustment, isLoading } = useClockLogAdjustment();
  
  const handleChange = async (newType: 'IN' | 'OUT') => {
    try {
      await updateAdjustment(clockLogId, { type: newType, justification: 'Corrección de tipo via selector inline' });
      setIsOpen(false);
      onChange?.();
    } catch (error) {
      console.error('Failed to update type:', error);
    }
  };

  return (
    <div className="relative">
      <button
        onClick={() => setIsOpen(!isOpen)}
        className="flex items-center gap-1 px-2 py-1 text-sm font-medium text-zinc-600 hover:text-zinc-800 bg-zinc-100 rounded transition-colors"
      >
        {currentType === 'IN' ? 'Entrada' : 'Salida'}
        <ChevronDownIcon className="w-4 h-4" />
      </button>
      
      {isOpen && (
        <div className="absolute top-full left-0 mt-1 bg-white border border-zinc-200 rounded-lg shadow-lg z-10">
          <button
            onClick={() => handleChange('IN')}
            disabled={isLoading || currentType === 'IN'}
            className="block w-full px-4 py-2 text-left text-sm hover:bg-zinc-50 disabled:opacity-50"
          >
            Entrada
          </button>
          <button
            onClick={() => handleChange('OUT')}
            disabled={isLoading || currentType === 'OUT'}
            className="block w-full px-4 py-2 text-left text-sm hover:bg-zinc-50 disabled:opacity-50"
          >
            Salida
          </button>
        </div>
      )}
    </div>
  );
}

import React, { useState } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { MarkConfidenceBadge } from './MarkConfidenceBadge';
import { MarkTypeSelector } from './MarkTypeSelector';
import { ClockLog } from '../types/clockLog';

interface DayMark {
  id: number;
  timestamp: string;
  type: 'IN' | 'OUT';
  confidence: 'HIGH' | 'MEDIUM' | 'LOW';
}

interface AuditDayRowProps {
  date: string;
  marks: DayMark[];
  hoursCalculated: number;
  isConfirmed: boolean;
  onConfirm: () => void;
}

export function AuditDayRow({ date, marks, hoursCalculated, isConfirmed, onConfirm }: AuditDayRowProps) {
  const [isExpanded, setIsExpanded] = useState(false);
  const dateObj = new Date(date);
  const dayName = dateObj.toLocaleDateString('es-CR', { weekday: 'long' });
  const dayNum = dateObj.getDate();
  const month = dateObj.toLocaleDateString('es-CR', { month: 'short' });

  return (
    <div className="border border-zinc-200 rounded-lg overflow-hidden">
      {/* Collapsed row - always visible */}
      <button
        onClick={() => setIsExpanded(!isExpanded)}
        className="w-full flex items-center justify-between px-4 py-3 bg-white hover:bg-zinc-50 transition-colors"
      >
        <div className="flex items-center gap-3">
          <ChevronDownIcon className={`w-5 h-5 text-zinc-400 transition-transform ${isExpanded ? 'rotate-180' : ''}`} />
          <div className="text-left">
            <div className="font-medium text-zinc-800">
              {dayName} {dayNum} {month}
            </div>
            <div className="text-sm text-zinc-500">
              {marks.length} marcas
            </div>
          </div>
        </div>
        
        <div className="flex items-center gap-4">
          {/* Hours always visible (D-11) */}
          <div className="text-right">
            <div className="font-semibold text-zinc-800">
              {hoursCalculated.toFixed(1)}h
            </div>
            <div className="text-xs text-zinc-500">calculadas</div>
          </div>
          
          {/* Confirmation status */}
          {isConfirmed && (
            <span className="inline-flex items-center gap-1 px-2 py-1 bg-green-100 text-green-700 rounded-full text-xs font-medium">
              <CheckIcon className="w-3 h-3" />
              Revisado
            </span>
          )}
        </div>
      </button>
      
      {/* Expanded content */}
      <AnimatePresence>
        {isExpanded && (
          <motion.div
            initial={{ height: 0, opacity: 0 }}
            animate={{ height: 'auto', opacity: 1 }}
            exit={{ height: 0, opacity: 0 }}
            className="border-t border-zinc-200 bg-zinc-50"
          >
            <div className="p-4 space-y-2">
              {marks.map((mark) => (
                <div
                  key={mark.id}
                  className="flex items-center justify-between px-3 py-2 bg-white rounded border border-zinc-200"
                >
                  <div className="flex items-center gap-3">
                    <MarkConfidenceBadge level={mark.confidence} />
                    <span className="font-mono text-sm">
                      {new Date(mark.timestamp).toLocaleTimeString('es-CR', { hour: '2-digit', minute: '2-digit' })}
                    </span>
                  </div>
                  <MarkTypeSelector
                    clockLogId={mark.id}
                    currentType={mark.type}
                  />
                </div>
              ))}
              
              {/* Confirm button (D-10) */}
              <button
                onClick={onConfirm}
                disabled={isConfirmed}
                className={`w-full py-2 rounded-lg font-medium transition-colors ${
                  isConfirmed
                    ? 'bg-zinc-100 text-zinc-400 cursor-not-allowed'
                    : 'bg-blue-500 text-white hover:bg-blue-600'
                }`}
              >
                {isConfirmed ? 'Día confirmado' : 'Confirmar día'}
              </button>
            </div>
          </motion.div>
        )}
      </AnimatePresence>
    </div>
  );
}
