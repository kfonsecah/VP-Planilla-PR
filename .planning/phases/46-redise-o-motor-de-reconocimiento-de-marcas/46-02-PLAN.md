```typescript
import { prisma } from '../lib/prisma';
import { CreateDayConfirmationInput } from '../schemas/DayConfirmationSchema';

export class DayConfirmationService {
  static async create(employeeId: number, date: string, userId: number, notes?: string) {
    return await prisma.vpgDayConfirmation.create({
      data: {
        employee_id: employeeId,
        confirmation_date: new Date(date),
        confirmed_by: userId,
        confirmation_notes: notes,
      },
    });
  }

  static async getByEmployee(employeeId: number, startDate?: string, endDate?: string) {
    return await prisma.vpgDayConfirmation.findMany({
      where: {
        employee_id: employeeId,
        ...(startDate && endDate ? {
          confirmation_date: {
            gte: new Date(startDate),
            lte: new Date(endDate),
          },
        } : {}),
      },
      orderBy: { confirmation_date: 'desc' },
    });
  }

  static async getById(id: number) {
    return await prisma.vpgDayConfirmation.findUnique({
      where: { confirmation_id: id },
    });
  }

  static async update(id: number, notes: string) {
    return await prisma.vpgDayConfirmation.update({
      where: { confirmation_id: id },
      data: { confirmation_notes: notes },
    });
  }

  static async delete(id: number) {
    return await prisma.vpgDayConfirmation.delete({
      where: { confirmation_id: id },
    });
  }
}
```

```typescript
import { Router } from 'express';
import { asyncHandler } from '../utils/asyncHandler';
import { AuthMiddleware } from '../middleware/AuthMiddleware';
import { TimeWindowService } from '../service/TimeWindowService';
import { CreateTimeWindowInput, UpdateTimeWindowInput, TimeWindowParams } from '../schemas/TimeWindowSchema';

const router = Router();

router.post('/', AuthMiddleware.verifyToken, asyncHandler(async (req, res) => {
  const data = CreateTimeWindowInput.parse(req.body);
  const result = await TimeWindowService.create(data);
  res.status(201).json(result);
}));

router.get('/', AuthMiddleware.verifyToken, asyncHandler(async (req, res) => {
  const companyId = req.user?.companyId;
  const result = await TimeWindowService.getAll(companyId!);
  res.json(result);
}));

router.get('/:id', AuthMiddleware.verifyToken, asyncHandler(async (req, res) => {
  const { id } = TimeWindowParams.parse(req.params);
  const result = await TimeWindowService.getById(id);
  if (!result) throw new Error('Ventana no encontrada');
  res.json(result);
}));

router.put('/:id', AuthMiddleware.verifyToken, asyncHandler(async (req, res) => {
  const { id } = TimeWindowParams.parse(req.params);
  const data = UpdateTimeWindowInput.parse(req.body);
  const result = await TimeWindowService.update(id, data);
  res.json(result);
}));

router.delete('/:id', AuthMiddleware.verifyToken, asyncHandler(async (req, res) => {
  const { id } = TimeWindowParams.parse(req.params);
  const result = await TimeWindowService.delete(id);
  res.json(result);
}));

export default router;
```
