import ExcelJS from 'exceljs';

export interface DetectedColumns {
  name?: number;      // Column index for employee name
  date?: number;     // Column index for date
  time?: number;     // Column index for time
  timestamp?: number; // Column index for combined datetime
}

export interface ColumnDetectionResult {
  columns: DetectedColumns;
  confidence: Record<keyof DetectedColumns, 'HIGH' | 'MEDIUM' | 'LOW'>;
  skippedRows: number[];
}

/**
 * Analyzes first 10 rows of Excel to detect column roles.
 * Uses pattern matching on cell content, not headers.
 */
export async function detectColumns(buffer: ArrayBuffer): Promise<ColumnDetectionResult> {
  const workbook = new ExcelJS.Workbook();
  await workbook.xlsx.load(buffer);
  const worksheet = workbook.getWorksheet(1);
  const rows: any[][] = [];
  
  worksheet.eachRow((row, rowNum) => {
    if (rowNum <= 10) rows.push(row.values as any[]);
  });

  const columns: DetectedColumns = {};
  const skippedRows: number[] = [];

  // Detect NAME column: contains name-like patterns
  for (let col = 0; col < rows[0].length; col++) {
    const values = rows.slice(1).map(r => r[col]?.toString() || '');
    const nameCount = values.filter(v => 
      /^[A-Za-zÁÉÍÓÚáéíóúÑñ\s]{3,}$/.test(v)
    ).length;
    if (nameCount >= 3) columns.name = col;
  }

  // Detect DATE column: date patterns
  for (let col = 0; col < rows[0].length; col++) {
    const values = rows.slice(1).map(r => r[col]?.toString() || '');
    const datePatterns = [
      /^\d{1,2}[\/\-]\d{1,2}[\/\-]\d{2,4}$/,
      /^\d{4}[\/\-]\d{1,2}[\/\-]\d{1,2}$/,
      /^\d{5,}$/, // Excel serial
    ];
    const dateCount = values.filter(v => datePatterns.some(p => p.test(v))).length;
    if (dateCount >= 3) columns.date = col;
  }

  // Detect TIME column: time patterns
  for (let col = 0; col < rows[0].length; col++) {
    const values = rows.slice(1).map(r => r[col]?.toString() || '');
    const timePatterns = [
      /^\d{1,2}:\d{2}(?::\d{2})?$/,
      /^\d{1,2}:\d{2}\s*[AaPp][Mm]$/,
      /^\d{4}$/, // Excel time serial
    ];
    const timeCount = values.filter(v => timePatterns.some(p => p.test(v))).length;
    if (timeCount >= 3) columns.time = col;
  }

  // Detect TIMESTAMP (combined date+time in single column)
  if (!columns.date || !columns.time) {
    for (let col = 0; col < rows[0].length; col++) {
      const values = rows.slice(1).map(r => r[col]?.toString() || '');
      const tsPatterns = [
        /^\d{1,2}[\/\-]\d{1,2}[\/\-]\d{2,4}\s+\d{1,2}:\d{2}/,
        /^\d{4}[\/\-]\d{1,2}[\/\-]\d{1,2}\s+T\d{1,2}:\d{2}/,
      ];
      const tsCount = values.filter(v => tsPatterns.some(p => p.test(v))).length;
      if (tsCount >= 3) columns.timestamp = col;
    }
  }

  // Build confidence map
  const confidence: any = {};
  const detectConfidence = (col: number | undefined, pool: number): 'HIGH' | 'MEDIUM' | 'LOW' => {
    if (col === undefined) return 'LOW';
    return pool >= 5 ? 'HIGH' : pool >= 3 ? 'MEDIUM' : 'LOW';
  };
  
  confidence.name = detectConfidence(columns.name, rows.filter(r => r[columns.name!]).length);
  confidence.date = detectConfidence(columns.date, rows.filter(r => r[columns.date!]).length);
  confidence.time = detectConfidence(columns.time, rows.filter(r => r[columns.time!]).length);
  confidence.timestamp = detectConfidence(columns.timestamp, rows.filter(r => r[columns.timestamp!]).length);

  return { columns, confidence, skippedRows };
}

export interface ParsedDateTime {
  date: string;  // YYYY-MM-DD
  time: string;  // HH:MM
  raw: string;
}

/**
 * Parses multiple date/time formats into structured output.
 * Supported:
 * - DD/MM/YYYY HH:MM
 * - YYYY-MM-DD HH:MM:SS
 * - Excel serial dates
 * - Combined timestamp column
 */
export function parseDateTime(dateValue: any, timeValue?: any): ParsedDateTime {
  let dateStr = '';
  let timeStr = '';
  
  // Handle Excel serial date (number)
  if (typeof dateValue === 'number') {
    const excelEpoch = new Date(1899, 11, 30);
    const parsed = new Date(excelEpoch.getTime() + dateValue * 86400000);
    dateStr = parsed.toISOString().split('T')[0];
  }
  // Handle string formats
  else if (typeof dateValue === 'string') {
    // Try DD/MM/YYYY
    let match = dateValue.match(/^(\d{1,2})[\/\-](\d{1,2})[\/\-](\d{2,4})$/);
    if (match) {
      const day = match[1].padStart(2, '0');
      const month = match[2].padStart(2, '0');
      const year = match[3].length === 2 ? '20' + match[3] : match[3];
      dateStr = `${year}-${month}-${day}`;
    }
    // Try YYYY-MM-DD
    match = dateValue.match(/^(\d{4})[\/\-](\d{1,2})[\/\-](\d{1,2})$/);
    if (match) {
      dateStr = `${match[1]}-${match[2].padStart(2, '0')}-${match[3].padStart(2, '0')}`;
    }
  }

  // Handle time
  if (timeValue) {
    if (typeof timeValue === 'number') {
      // Excel time serial (decimal part of day)
      const hours = Math.floor(timeValue * 24);
      const mins = Math.floor((timeValue * 24 * 60) % 60);
      timeStr = `${hours.toString().padStart(2, '0')}:${mins.toString().padStart(2, '0')}`;
    } else if (typeof timeValue === 'string') {
      // HH:MM:SS or HH:MM AM/PM
      const match = timeValue.match(/^(\d{1,2}):(\d{2})/);
      if (match) {
        timeStr = `${match[1].padStart(2, '0')}:${match[2]}`;
      }
    }
  }

  return {
    date: dateStr || new Date().toISOString().split('T')[0],
    time: timeStr || '00:00',
    raw: dateValue + ' ' + timeValue,
  };
}

import { TimeWindow } from '../services/timeWindowService';

export type ConfidenceLevel = 'HIGH' | 'MEDIUM' | 'LOW';

export interface ClassificationResult {
  type: 'IN' | 'OUT' | 'UNDETERMINED';
  confidence: ConfidenceLevel;
  windowName?: string;
}

/**
 * Classifies a timestamp as IN or OUT based on configured time windows.
 * Returns confidence based on how close to window center.
 */
export function classifyByTimeWindow(
  time: string, // HH:MM format
  windows: TimeWindow[]
): ClassificationResult {
  const [hours, mins] = time.split(':').map(Number);
  const minutesSinceMidnight = hours * 60 + mins;

  let bestMatch: { window: TimeWindow; distance: number } | null = null;

  for (const window of windows) {
    const [startHour, startMin] = window.startHour.split(':').map(Number);
    const [endHour, endMin] = window.endHour.split(':').map(Number);
    const startMins = startHour * 60 + startMin;
    const endMins = endHour * 60 + endMin;

    if (minutesSinceMidnight >= startMins && minutesSinceMidnight <= endMins) {
      const center = (startMins + endMins) / 2;
      const distance = Math.abs(minutesSinceMidnight - center);
      
      if (!bestMatch || distance < bestMatch.distance) {
        bestMatch = { window, distance };
      }
    }
  }

  if (!bestMatch) {
    return { type: 'UNDETERMINED', confidence: 'LOW' };
  }

  const confidence = bestMatch.distance < 15 ? 'HIGH' : 
                  bestMatch.distance < 45 ? 'MEDIUM' : 'LOW';
  
  const type = bestMatch.window.type.includes('ENTRY') ? 'IN' : 'OUT';

  return {
    type,
    confidence,
    windowName: bestMatch.window.name,
  };
}
